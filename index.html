function peak(arr) {
  for (let i = 1; i < arr.length - 1; i++) {
    let leftSum = arr.slice(0, i)
      .reduce((acc, val) => acc + val);
    let rightSum = arr.slice(i + 1)
      .reduce((acc, val) => acc + val);
    if (leftSum === rightSum) return i;
  }
  return -1;
}
